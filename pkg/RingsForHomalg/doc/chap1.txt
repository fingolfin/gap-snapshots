  
  [1X1 Introduction[0X
  
  This  package is part of the [5Xhomalg[0m project [hpa10]. The role of the package
  is described in the manual of the [5Xhomalg[0m package.
  
  
  [1X1.1 Ring Constructions for Supported External Computer Algebra Systems[0X
  
  Here are some of the supported ring constructions:
  
  
  [1X1.1-1 external [5XGAP[1X[0X
  
  [4X---------------------------  Example  ----------------------------[0X
    [4Xgap> ZZ := HomalgRingOfIntegersInExternalGAP( );[0X
    [4XZ[0X
    [4Xgap> Display( ZZ );[0X
    [4X<An external ring residing in the CAS GAP>[0X
    [4Xgap> F2 := HomalgRingOfIntegersInExternalGAP( 2, ZZ );[0X
    [4XGF(2)[0X
    [4Xgap> Display( F2 );[0X
    [4X<An external ring residing in the CAS GAP>[0X
  [4X------------------------------------------------------------------[0X
  
  [10XF2 := HomalgRingOfIntegersInExternalGAP( 2 )[0m would launch another GAP.
  
  [4X---------------------------  Example  ----------------------------[0X
    [4Xgap> Z4 := HomalgRingOfIntegersInExternalGAP( 4, ZZ );[0X
    [4XZ/4Z[0X
    [4Xgap> Display( Z4 );[0X
    [4X<An external ring residing in the CAS GAP>[0X
    [4Xgap> Z_4 := HomalgRingOfIntegersInExternalGAP( ZZ ) / 4;[0X
    [4XZ/( 4 )[0X
    [4Xgap> Display( Z_4 );[0X
    [4X<A residue class ring>[0X
    [4Xgap> Q := HomalgFieldOfRationalsInExternalGAP( ZZ );[0X
    [4XQ[0X
    [4Xgap> Display( Q );[0X
    [4X<An external ring residing in the CAS GAP>[0X
  [4X------------------------------------------------------------------[0X
  
  
  [1X1.1-2 [5XSingular[1X[0X
  
  [4X---------------------------  Example  ----------------------------[0X
    [4Xgap> F2 := HomalgRingOfIntegersInSingular( 2 );[0X
    [4XGF(2)[0X
    [4Xgap> Display( F2 );[0X
    [4X<An external ring residing in the CAS Singular>[0X
    [4Xgap> F2s := HomalgRingOfIntegersInSingular( 2, "s" ,F2 );[0X
    [4XGF(2)(s)[0X
    [4Xgap> Display( F2s );[0X
    [4X<An external ring residing in the CAS Singular>[0X
    [4Xgap> ZZ := HomalgRingOfIntegersInSingular( F2 );[0X
    [4XZ[0X
    [4Xgap> Display( ZZ );[0X
    [4X<An external ring residing in the CAS Singular>[0X
    [4Xgap> ZZs := HomalgRingOfIntegersInSingular( "s", F2 );[0X
    [4XZ(s)[0X
    [4Xgap> Display( ZZs );[0X
    [4X<An external ring residing in the CAS Singular>[0X
    [4Xgap> Q := HomalgFieldOfRationalsInSingular( F2 );[0X
    [4XQ[0X
    [4Xgap> Display( Q );[0X
    [4X<An external ring residing in the CAS Singular>[0X
    [4Xgap> Qs := HomalgFieldOfRationalsInSingular( "s", F2 );[0X
    [4XQ(s)[0X
    [4Xgap> Display( Qs );[0X
    [4X<An external ring residing in the CAS Singular>[0X
    [4Xgap> Qi := HomalgFieldOfRationalsInSingular( "i", "i^2+1", Q );[0X
    [4XQ[i]/(i^2+1)[0X
    [4Xgap> Display( Qi );[0X
    [4X<An external ring residing in the CAS Singular>[0X
  [4X------------------------------------------------------------------[0X
  
  [10XQ := HomalgFieldOfRationalsInSingular( )[0m would launch another Singular.
  
  [4X---------------------------  Example  ----------------------------[0X
    [4Xgap> F2xyz := F2 * "x,y,z";[0X
    [4XGF(2)[x,y,z][0X
    [4Xgap> Display( F2xyz );[0X
    [4X<An external ring residing in the CAS Singular>[0X
    [4Xgap> F2sxyz := F2s * "x,y,z";[0X
    [4XGF(2)(s)[x,y,z][0X
    [4Xgap> Display( F2sxyz );[0X
    [4X<An external ring residing in the CAS Singular>[0X
    [4Xgap> F2xyzw := F2xyz * "w";[0X
    [4XGF(2)[x,y,z][w][0X
    [4Xgap> Display( F2xyzw );[0X
    [4X<An external ring residing in the CAS Singular>[0X
    [4Xgap> F2sxyzw := F2sxyz * "w";[0X
    [4XGF(2)(s)[x,y,z][w][0X
    [4Xgap> Display( F2sxyzw );[0X
    [4X<An external ring residing in the CAS Singular>[0X
    [4Xgap> ZZxyz := ZZ * "x,y,z";[0X
    [4XZ[x,y,z][0X
    [4Xgap> Display( ZZxyz );[0X
    [4X<An external ring residing in the CAS Singular>[0X
    [4Xgap> ZZsxyz := ZZs * "x,y,z";[0X
    [4XZ(s)[x,y,z][0X
    [4Xgap> Display( ZZsxyz );[0X
    [4X<An external ring residing in the CAS Singular>[0X
    [4Xgap> ZZxyzw := ZZxyz * "w";[0X
    [4XZ[x,y,z][w][0X
    [4Xgap> Display( ZZxyzw );[0X
    [4X<An external ring residing in the CAS Singular>[0X
    [4Xgap> ZZsxyzw := ZZsxyz * "w";[0X
    [4XZ(s)[x,y,z][w][0X
    [4Xgap> Display( ZZsxyzw );[0X
    [4X<An external ring residing in the CAS Singular>[0X
    [4Xgap> Qxyz := Q * "x,y,z";[0X
    [4XQ[x,y,z][0X
    [4Xgap> Display( Qxyz );[0X
    [4X<An external ring residing in the CAS Singular>[0X
    [4Xgap> Qsxyz := Qs * "x,y,z";[0X
    [4XQ(s)[x,y,z][0X
    [4Xgap> Display( Qsxyz );[0X
    [4X<An external ring residing in the CAS Singular>[0X
    [4Xgap> Qixyz := Qi * "x,y,z";[0X
    [4X(Q[i]/(i^2+1))[x,y,z][0X
    [4Xgap> Display( Qixyz );[0X
    [4X<An external ring residing in the CAS Singular>[0X
    [4Xgap> Qxyzw := Qxyz * "w";[0X
    [4XQ[x,y,z][w][0X
    [4Xgap> Display( Qxyzw );[0X
    [4X<An external ring residing in the CAS Singular>[0X
    [4Xgap> Qsxyzw := Qsxyz * "w";[0X
    [4XQ(s)[x,y,z][w][0X
    [4Xgap> Display( Qsxyzw );[0X
    [4X<An external ring residing in the CAS Singular>[0X
    [4Xgap> Dxyz := RingOfDerivations( Qxyz, "Dx,Dy,Dz" );[0X
    [4XQ[x,y,z]<Dx,Dy,Dz>[0X
    [4Xgap> Display( Dxyz );[0X
    [4X<An external ring residing in the CAS Singular>[0X
    [4Xgap> Exyz := ExteriorRing( Qxyz, "e,f,g" );[0X
    [4XQ{e,f,g}[0X
    [4Xgap> Display( Exyz );[0X
    [4X<An external ring residing in the CAS Singular>[0X
    [4Xgap> Dsxyz := RingOfDerivations( Qsxyz, "Dx,Dy,Dz" );[0X
    [4XQ(s)[x,y,z]<Dx,Dy,Dz>[0X
    [4Xgap> Display( Dsxyz );[0X
    [4X<An external ring residing in the CAS Singular>[0X
    [4Xgap> Esxyz := ExteriorRing( Qsxyz, "e,f,g" );[0X
    [4XQ(s){e,f,g}[0X
    [4Xgap> Display( Esxyz );[0X
    [4X<An external ring residing in the CAS Singular>[0X
    [4Xgap> Dixyz := RingOfDerivations( Qixyz, "Dx,Dy,Dz" );[0X
    [4X(Q[i]/(i^2+1))[x,y,z]<Dx,Dy,Dz>[0X
    [4Xgap> Display( Dixyz );[0X
    [4X<An external ring residing in the CAS Singular>[0X
    [4Xgap> Eixyz := ExteriorRing( Qixyz, "e,f,g" );[0X
    [4X(Q[i]/(i^2+1)){e,f,g}[0X
    [4Xgap> Display( Eixyz );[0X
    [4X<An external ring residing in the CAS Singular>[0X
